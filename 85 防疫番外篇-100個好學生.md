## 1-3. 程式 (m01.py)

``` python
#---------------------
# 匯入模組
#---------------------
import pygame
from random import randint

#=====================
# 好學生類別
#=====================
class Student(pygame.sprite.Sprite):
    # 建構元(x, y, 方向, 遲延時間)
    def __init__(self, x, y, direction, delay):
        super().__init__()
        r=randint(1, 49)
        fname = './images/a_' + str(r) + '.png'
        self.image = pygame.image.load(fname).convert_alpha()
        self.rect = self.image.get_rect()
        self.x = x
        self.y = y
        self.direction = direction
        self.delay = delay   
        self.frame = 0
        self.cnt = 0            

    # 移動
    def move(self):  
        # 圖片位置       
        self.rect.topleft = (self.x, self.y)

        # 是否更新下一張圖
        self.cnt+=1
        if self.cnt < self.delay:
            return
        else:
            self.cnt = 0

        # 更新下一張圖
        self.frame+=1
        if self.frame>3:
            self.frame=0

            # 是否轉方向
            if randint(1, 5)==5:
                x=randint(0, 3)

                if x==0:
                    self.turn_down()
                elif x==1:
                    self.turn_left()
                elif x==2:
                    self.turn_right()
                elif x==3:
                    self.turn_up()

        # 如果向下走
        if self.direction==0:
            self.y+=4

        # 如果向左走
        if self.direction==1:
            self.x-=4

        # 如果向右走
        if self.direction==2:
            self.x+=4  

        # 如果向上走
        if self.direction==3:
            self.y-=4                     

    # 向左走
    def turn_left(self):
        self.direction = 1

    # 向右走
    def turn_right(self):
        self.direction = 2

    # 向上走
    def turn_up(self):
        self.direction = 3

    # 向下走
    def turn_down(self):
        self.direction = 0  

    # 取得顯示子圖的左上角(x, y)及(寬度:32, 高度:48)
    def getFrame(self): 
        return self.frame*32, self.direction*48, 32, 48
```


## 1-4. 程式 (p01.py)

``` python
#---------------------
# 匯入模組
#---------------------
import sys
import pygame
from pygame.locals import QUIT
from random import randint

#---------------------
# 匯入[好學生]類別
#---------------------
from ntub.m01 import Student

#=====================
# 主程式
#=====================
# 啟動pygame
pygame.init()

# 設定畫面尺寸及標題
screen_surface = pygame.display.set_mode((800, 600))
pygame.display.set_caption('資一甲好學生大亂鬥')

# 產生100個好學生物件(座標x, 座標y, 方向, 更新圖片時間)    
students = []

for k in range(100):
    students.append(Student(randint(300, 600), randint(200, 400), randint(0, 3), randint(10, 20)))

# 產生時間物件, 用來設定切換畫面頻率
clock = pygame.time.Clock()

# 遊戲主流程, 無窮迴圈
while True:
    # 偵測事件
    for event in pygame.event.get():
        # 結束
        if event.type == QUIT:
            pygame.quit()
            sys.exit()               
    
    # 背景顏色, 也清除畫面
    screen_surface.fill((200, 200, 200))

    # 好學生移動 
    for stu in students:          
        stu.move()

    # 排序, y值愈小者, 排前面, 先畫
    students = sorted(students, key=lambda s: s.y)    

    # 畫好學生
    for stu in students:
        screen_surface.blit(stu.image, stu.rect, stu.getFrame())

    # 更新畫面
    pygame.display.update()

    # 遊戲速率(每秒畫面更新次數)
    clock.tick(100)
```
