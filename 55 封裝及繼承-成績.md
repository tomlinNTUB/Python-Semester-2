## 2-2. 類別 (m50.py)

``` python
#-------------------------
# 成績 (父類別)
#-------------------------
class Score():
    # 學號, 國文, 英文, 數學
    def __init__(self, no, chi, eng, mat):
        self.no = no
        self.chi = chi
        self.eng = eng
        self.mat = mat

    # 總分
    def total(self):
        return self.chi + self.eng + self.mat

    # 平均
    def average(self):
        return self.total() / 3

#-------------------------
# 體育保送成績 (子類別)
#-------------------------
class PEscore(Score):
    # 學號, 國文, 英文, 數學, 體育
    def __init__(self, no, chi, eng, mat, sports):
        super().__init__(no, chi, eng, mat)
        self.sports = sports

    # 總分
    def total(self):
        if self.sports == 'A+':
            tot = 100
        elif self.sports == 'A':
            tot = 90
        else:
            tot = 80

        return super().total() + tot

    # 平均
    def average(self):
        return self.total() / 4
```
