``` python
#---------------------
# 匯入模組
#---------------------
import pygame
from random import randint
import math

#=====================
# 好蜘蛛類別
#=====================
class Spider(pygame.sprite.Sprite):
    # 好蜘蛛圖片
    img = None 

    # 建構元(x, y, 方向, 遲延時間)
    def __init__(self, x, y, direction, delay):
        super().__init__()
        if Spider.img is None:
            Spider.img = pygame.image.load('./images/spider.png').convert_alpha()
        self.image = Spider.img
        self.rect = self.image.get_rect()
        self.x = x
        self.y = y
        self.direction = direction
        self.delay = delay   
        self.frame = 0
        self.cnt = 0            

    # 移動
    def move(self):  
        # 圖片位置       
        self.rect.topleft = (self.x, self.y)

        # 是否更新下一張圖
        self.cnt+=1
        if self.cnt < self.delay:
            return
        else:
            self.cnt = 0

        # 更新下一張圖
        self.frame+=1
        if self.frame>7:
            self.frame=0

            # 是否轉方向
            if randint(1, 5)==5:
                x=randint(-1, 1)

                self.direction+=x
                if self.direction<0:
                    self.direction=7

                if self.direction>7:
                    self.direction=0

        # 計算移動後位置
        sl = 1.5
        ssl = sl / math.sqrt(2)

        # 如果向左走        
        if self.direction==0:
            self.y+=0
            self.x-=sl

        # 如果向左上走
        if self.direction==1:
            self.y-=ssl
            self.x-=ssl

        # 如果向上走
        if self.direction==2:
            self.y-=sl
            self.x-=0 

        # 如果向右上走
        if self.direction==3:
            self.y-=ssl
            self.x+=ssl     

        # 如果向右走        
        if self.direction==4:
            self.y+=0
            self.x+=sl

        # 如果向右下走
        if self.direction==5:
            self.y+=ssl
            self.x+=ssl

        # 如果向下走
        if self.direction==6:
            self.y+=sl
            self.x-=0 

        # 如果向左下走
        if self.direction==7:
            self.y+=ssl
            self.x-=ssl                            


    # 取得顯示子圖的左上角(x, y)及(寬度:128, 高度:128)
    def getFrame(self): 
        return self.frame*128, self.direction*128, 128, 128

class SpiderGG(pygame.sprite.Sprite):
    # 好蜘蛛圖片
    img = None 

    # 建構元(x, y, 方向, 遲延時間)
    def __init__(self, x, y, direction, delay):
        super().__init__()
        if SpiderGG.img is None:
            SpiderGG.img = pygame.image.load('./images/spiderGG.png').convert_alpha()
        self.image = SpiderGG.img
        self.rect = self.image.get_rect()
        self.x = x
        self.y = y
        self.direction = direction
        self.delay = delay   
        self.frame = 0
        self.cnt = 0  
        self.back = randint(10, 20)          

    # 移動
    def move(self):  
        # 圖片位置       
        self.rect.topleft = (self.x, self.y)

        # 是否更新下一張圖
        self.cnt+=1
        if self.cnt < self.delay:
            return
        else:
            self.cnt = 0

        # 更新下一張圖
        self.frame+=1
        if self.frame>7:
            self.frame=7

        # 計算移動後位置                              
        self.y -= self.back
        self.back -= 1

        if self.back < 0:
            self.back = 0

    # 取得顯示子圖的左上角(x, y)及(寬度:128, 高度:128)
    def getFrame(self): 
        return self.frame*128, self.direction*128, 128, 128
```

## 1-4. 程式 (p01.py)

``` python
#---------------------
# 匯入模組
#---------------------
import sys
import math
from random import randint
import pygame
from pygame.locals import QUIT, KEYDOWN, MOUSEMOTION, MOUSEBUTTONDOWN, K_UP, K_DOWN, K_LEFT, K_RIGHT, K_SPACE, K_ESCAPE

#---------------------
# 匯入[好蜘蛛]類別
#---------------------
from ntub.m01 import Spider, SpiderGG, Explosion

# 設定畫布顯示位置
import os
os.environ['SDL_VIDEO_WINDOW_POS'] = '200,30'

#=====================
# 主程式
#=====================
# 啟動pygame
pygame.init()

# 設定畫面尺寸及標題
screen_surface = pygame.display.set_mode((700, 700))
pygame.display.set_caption('進擊的好蜘蛛')

# 產生時間物件, 用來設定切換畫面頻率
clock = pygame.time.Clock()

# 剩餘秒數
sec = 99

# 鍵盤動作
key = ''

# 滑鼠位置及按鈕動作
mouse_x = 0
mouse_y = 0
mouse_action = ''

# 產生100個好蜘蛛物件(座標x, 座標y, 方向, 更新圖片時間)    
spiders = []
for i in range(100):
    spiders.append(Spider(350-64, 300, randint(0, 7), randint(2, 5)))

# 產生好蜘蛛GG物件(座標x, 座標y, 方向, 更新圖片時間)    
spidersGG = []

# 子彈
bullets = []

# 爆炸
explosions = []

# 子彈間隔
bulletCnt = 0

# 遊戲主流程, 無窮迴圈
while True:
    # 偵測事件
    for event in pygame.event.get():
        # 結束
        if event.type == QUIT:
            pygame.quit()
            sys.exit()               
    
        # 如果按鍵
        if event.type == KEYDOWN:
            if event.key == K_UP:
                key = 'Up'
            elif event.key == K_DOWN:
                key = 'Down'
            elif event.key == K_LEFT:
                key = 'Left'
            elif event.key == K_RIGHT:
                key = 'Right'
            elif event.key == K_SPACE:
                key = 'Space'

        # 取得滑鼠位置
        if event.type == MOUSEMOTION:
            mpos = pygame.mouse.get_pos()
            mouse_x = mpos[0]
            mouse_y = mpos[1]

        # 取得滑鼠按鈕動作
        if event.type == MOUSEBUTTONDOWN:
            pressed_array = pygame.mouse.get_pressed()
            for index in range(len(pressed_array)):
                if pressed_array[index]:
                    if index == 0:
                        mouse_action = 'left'
                    elif index == 1:
                        mouse_action = 'wheel'
                    elif index == 2:
                        mouse_action = 'right'

    # 畫地圖
    screen_surface.fill((187, 67, 0))

    # 加入子彈
    if mouse_action == 'left':
        bullets.append([mouse_x, 680])
        mouse_action = ''
        bulletCnt+=1

    if mouse_action == 'right' and bulletCnt>5:
        bullets.append([mouse_x-20, 680])
        bullets.append([mouse_x, 680])
        bullets.append([mouse_x+20, 680])
        mouse_action = ''
        bulletCnt-=5
    
    # 畫文字(剩餘秒數)
    font = pygame.font.SysFont('arial black', 280)
    txt = str(100-len(spiders))
    text = font.render(txt, True, (255, 255, 255))
    screen_surface.blit(text, (80+(3-len(txt))*75, 0))

    # 好蜘蛛GG移動
    for s in spidersGG:          
        s.move()

    # 排序, y值愈小者, 排前面, 先畫
    spidersGG = sorted(spidersGG, key=lambda s: s.y)    

    # 畫好蜘蛛GG 
    for s in spidersGG:
        screen_surface.blit(s.image, s.rect, s.getFrame())

    # 好蜘蛛移動
    for s in spiders:          
        s.move()

    # 排序, y值愈小者, 排前面, 先畫
    spiders = sorted(spiders, key=lambda s: s.y)    

    # 畫好蜘蛛 
    for s in spiders:
        screen_surface.blit(s.image, s.rect, s.getFrame())

    # 畫好子彈
    for bullet in bullets:
        pygame.draw.rect(screen_surface, (255, 0, 0), pygame.Rect(bullet[0], bullet[1], 8, 20))
        pygame.draw.rect(screen_surface, (255, 255, 0), pygame.Rect(bullet[0]+3, bullet[1]+3, 2, 14))
        
        # 檢查好子彈是否擊中好蜘蛛
        for spider in spiders:
            bx = bullet[0]+4
            by = bullet[1]+10
            sx = spider.x + 64
            sy = spider.y + 64

            # 如果撃中好蜘蛛, 增加好蜘蛛GG, 回收好蜘蛛
            if math.sqrt((bx-sx)**2 + (by-sy)**2) < 40:
                spidersGG.append(SpiderGG(spider.x, spider.y, spider.direction, 2))
                spiders.remove(spider)
                explosions.append(Explosion(sx, sy, randint(2, 6)))
                bullet[1] = -100

        # 檢查好子彈是否擊中好蜘蛛GG
        for spiderGG in spidersGG:
            bx = bullet[0]+4
            by = bullet[1]+10
            sx = spiderGG.x + 64
            sy = spiderGG.y + 64

            # 如果撃中好蜘蛛GG
            if math.sqrt((bx-sx)**2 + (by-sy)**2) < 20:                
                spiderGG.back = randint(25, 35)

        # 子彈向上飛   
        bullet[1] -= 10    

    # 回收子彈
    for bullet in bullets:
        if bullet[1] < -100:
            bullets.remove(bullet)

    # 畫好爆炸
    for explosion in explosions:
        explosion.move()
        screen_surface.blit(explosion.image, explosion.rect, explosion.getFrame())

        # 畫完就回收
        if explosion.frame>15:
            explosions.remove(explosion)

    # 更新畫面
    pygame.display.update()

    # 遊戲速率(每秒畫面更新次數)
    clock.tick(100)
```
