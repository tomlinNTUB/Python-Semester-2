## 1-3. 程式 (m01.py)

``` python
#---------------------
# 匯入模組
#---------------------
import pygame

#=====================
# 好學生類別
#=====================
class Student(pygame.sprite.Sprite):
    # 建構元(x, y, 方向, 遲延時間)
    def __init__(self, x, y):
        super().__init__()
        self.image = pygame.image.load('./images/a_1.png').convert_alpha()
        self.rect = self.image.get_rect()
        self.x = x
        self.y = y   

    # 取得顯示子圖的左上角(x, y)及(寬度:32, 高度:48)
    def getFrame(self): 
        return 0, 0, 32, 48
```


## 1-4. 程式 (p01.py)

``` python
#---------------------
# 匯入模組
#---------------------
import sys
import pygame
from pygame.locals import QUIT
from random import randint

#---------------------
# 匯入[好學生]類別
#---------------------
from ntub.m01 import Student

#=====================
# 主程式
#=====================
# 啟動pygame
pygame.init()

# 設定畫面尺寸及標題
screen_surface = pygame.display.set_mode((800, 600))
pygame.display.set_caption('好學生大亂鬥')

# 產生好學生物件(座標x, 座標y)    
stu = Student(300, 200)

# 產生時間物件, 用來設定切換畫面頻率
clock = pygame.time.Clock()

# 遊戲主流程, 無窮迴圈
while True:
    # 偵測事件
    for event in pygame.event.get():
        # 結束
        if event.type == QUIT:
            pygame.quit()
            sys.exit()               
    
    # 背景顏色, 也清除畫面
    screen_surface.fill((200, 200, 200))

    # 在畫布上畫好學生
    screen_surface.blit(stu.image, stu.rect, stu.getFrame())

    # 更新畫面
    pygame.display.update()

    # 遊戲速率(每秒畫面更新次數)
    clock.tick(100)
```
