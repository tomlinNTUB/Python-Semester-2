``` python
#---------------------
# 匯入模組
#---------------------
import pygame
from random import randint

#=====================
# 好學生類別
#=====================
class Student(pygame.sprite.Sprite):
    # 建構元(x, y, 方向, 遲延時間)
    def __init__(self, x, y, direction, delay):
        super().__init__()
        r=randint(1, 49)
        fname = './images/a_' + str(r) + '.png'
        self.image = pygame.image.load(fname).convert_alpha()
        self.rect = self.image.get_rect()
        self.x = x
        self.y = y
        self.direction = direction
        self.delay = delay   
        self.frame = 0
        self.cnt = 0 
        self.step = 0     
        self.offsetX = 0
        self.offsetY = 0  
        self.next = self.direction   
        self.still = False 

    # 移動
    def move(self):  
        # 圖片位置       
        self.rect.topleft = (30+self.x*32+self.offsetX, 30+self.y*32-20+self.offsetY)

        # 是否更新下一張圖
        self.cnt+=1
        if self.cnt < self.delay:
            return
        else:
            self.cnt = 0

        # 更改方向
        if randint(1, 20)==1:
            x = randint(1, 9)
            if x in (1, 2):
                self.turn_down()
            elif x in (3, 4):
                self.turn_up()
            elif x in (5, 6):
                self.turn_left()
            elif x in (7, 8):
                self.turn_right()
            elif x == 9:
                self.turn_still()             

        # 如果不是停止狀態
        if not (self.still and self.step==0):
            # 更新下一張圖
            self.frame+=1
            if self.frame>3:
                self.frame=0            

            self.step+=1

            # 如果向下走
            if self.direction==0:
                self.offsetX=0
                self.offsetY+=4

            # 如果向左走
            if self.direction==1:
                self.offsetX-=4
                self.offsetY=0

            # 如果向右走
            if self.direction==2:
                self.offsetX+=4
                self.offsetY=0

            # 如果向上走
            if self.direction==3:
                self.offsetX=0
                self.offsetY-=4                    

            if self.step==8:
                self.step=0
                self.offsetX=0
                self.offsetY=0

                if self.direction==0:
                    self.y+=1

                if self.direction==1:
                    self.x-=1

                if self.direction==2:
                    self.x+=1

                if self.direction==3:
                    self.y-=1    

                self.direction = self.next                            
            
    # 向左走
    def turn_left(self):
        if self.still:
            self.direction = 1

        self.still = False
        self.next = 1

    # 向右走
    def turn_right(self):
        if self.still:
            self.direction = 2

        self.still = False
        self.next = 2

    # 向上走
    def turn_up(self):
        if self.still:
            self.direction = 3

        self.still = False
        self.next = 3

    # 向下走
    def turn_down(self):
        if self.still:            
            self.direction = 0

        self.still = False
        self.next = 0  

    # 停止
    def turn_still(self):
        if self.still:
            self.still = False
        else:            
            self.next = self.direction
            self.still = True         

    # 取得顯示子圖的左上角(x, y)及(寬度:32, 高度:48)
    def getFrame(self): 
        return self.frame*32, self.direction*48, 32, 48
```

## 1-4. 程式 (p01.py)

``` python
#---------------------
# 匯入模組
#---------------------
import sys
import pygame
from pygame.locals import QUIT, KEYDOWN, K_UP, K_DOWN, K_LEFT, K_RIGHT, K_SPACE
from random import randint

#---------------------
# 匯入[好學生]類別
#---------------------
from ntub.m01 import Student, Cat, Map, Ghost

#=====================
# 主程式
#=====================
# 啟動pygame
pygame.init()

# 設定畫面尺寸及標題
screen_surface = pygame.display.set_mode((700, 700))
pygame.display.set_caption('好學生大亂鬥之猫的大逃殺')

# 好學生清單 
students = []

# 產生50個好學生(座標x, 座標y, 方向, 更新圖片時間)
for i in range(0, 49):
    students.append(Student(randint(9, 11), randint(9, 11), randint(0, 3), randint(8, 18)))

# 產生1隻好猫(座標x, 座標y, 方向, 更新圖片時間)
cat = Cat(0, 0, 0, 3)

# 被吃掉的人數
eaten = 0

# 好鬼魂清單
ghosts = []

# 產生地圖物件(座標x, 座標y, 方向, 更新圖片時間)    
map = Map(30, 30)

# 產生時間物件, 用來設定切換畫面頻率
clock = pygame.time.Clock()

# 遊戲主流程, 無窮迴圈
while True:
    # 偵測事件
    for event in pygame.event.get():
        # 結束
        if event.type == QUIT:
            pygame.quit()
            sys.exit()       

        # 如果按鍵
        if event.type == KEYDOWN:
            if event.key == K_UP:
                cat.turn_up() 
            elif event.key == K_DOWN:
                cat.turn_down()
            elif event.key == K_LEFT:
                cat.turn_left()  
            elif event.key == K_RIGHT:
                cat.turn_right()    
            elif event.key == K_SPACE:
                cat.turn_still()  

    # 清除被吃掉的好學生, 變成好鬼魂
    for stu in students:
        if stu.x==cat.x and stu.y==cat.y:
            students.remove(stu)
            eaten+=1
            ghosts.append(Ghost(stu.x, stu.y, stu.direction, stu.delay))

    # 背景顏色, 也清除畫面
    screen_surface.fill((20, 20, 20))

    # 畫地圖
    screen_surface.blit(map.image, map.rect, map.getFrame())

    # 好鬼魂移動
    for ghost in ghosts:          
        ghost.move()

    # 畫好鬼魂
    for ghost in ghosts:
        screen_surface.blit(ghost.image, ghost.rect, ghost.getFrame())

    # 畫被吃掉的好學生數量
    font = pygame.font.SysFont('arial', 80)
    text = font.render(str(eaten), True, (255, 255, 255))
    screen_surface.blit(text, (600, 10))

    # 好學生移動 
    for stu in students:          
        stu.move()

    # 排序, y值愈小者, 排前面, 先畫
    students = sorted(students, key=lambda s: s.y)    

    # 畫好學生
    for stu in students:
        screen_surface.blit(stu.image, stu.rect, stu.getFrame())

    # 好猫移動 
    cat.move()

    # 畫好猫
    screen_surface.blit(cat.image, cat.rect, cat.getFrame())        

    # 更新畫面
    pygame.display.update()

    # 遊戲速率(每秒畫面更新次數)
    clock.tick(100)
```
