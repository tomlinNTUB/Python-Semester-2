## 1-3. 程式 (m01.py)

``` python
#---------------------
# 匯入模組
#---------------------
import pygame
from random import randint

#=====================
# 好學生類別
#=====================
class Student(pygame.sprite.Sprite):
    # 建構元(x, y, 方向, 遲延時間)
    def __init__(self, x, y, direction, delay):
        super().__init__()
        r=randint(1, 49)
        fname = './images/a_' + str(r) + '.png'
        self.image = pygame.image.load(fname).convert_alpha()
        self.rect = self.image.get_rect()
        self.x = x
        self.y = y
        self.direction = direction
        self.delay = delay   
        self.frame = 0
        self.cnt = 0 
        self.step = 0     
        self.offsetX = 0
        self.offsetY = 0  
        self.next = self.direction   
        self.still = False 

    # 移動
    def move(self):  
        # 圖片位置       
        self.rect.topleft = (30+self.x*32+self.offsetX, 30+self.y*32-20+self.offsetY)

        # 是否更新下一張圖
        self.cnt+=1
        if self.cnt < self.delay:
            return
        else:
            self.cnt = 0

        # 如果不是停止狀態
        if not (self.still and self.step==0):
            # 更新下一張圖
            self.frame+=1
            if self.frame>3:
                self.frame=0            

            self.step+=1

            # 如果向下走
            if self.direction==0:
                self.offsetX=0
                self.offsetY+=4

            # 如果向左走
            if self.direction==1:
                self.offsetX-=4
                self.offsetY=0

            # 如果向右走
            if self.direction==2:
                self.offsetX+=4
                self.offsetY=0

            # 如果向上走
            if self.direction==3:
                self.offsetX=0
                self.offsetY-=4                    

            if self.step==8:
                self.step=0
                self.offsetX=0
                self.offsetY=0

                if self.direction==0:
                    self.y+=1

                if self.direction==1:
                    self.x-=1

                if self.direction==2:
                    self.x+=1

                if self.direction==3:
                    self.y-=1    

                self.direction = self.next                            
            
    # 向左走
    def turn_left(self):
        if self.still:
            self.direction = 1

        self.still = False
        self.next = 1

    # 向右走
    def turn_right(self):
        if self.still:
            self.direction = 2

        self.still = False
        self.next = 2

    # 向上走
    def turn_up(self):
        if self.still:
            self.direction = 3

        self.still = False
        self.next = 3

    # 向下走
    def turn_down(self):
        if self.still:            
            self.direction = 0

        self.still = False
        self.next = 0  

    # 停止
    def turn_still(self):
        if self.still:
            self.still = False
        else:            
            self.next = self.direction
            self.still = True         

    # 取得顯示子圖的左上角(x, y)及(寬度:32, 高度:48)
    def getFrame(self): 
        return self.frame*32, self.direction*48, 32, 48                                   


#=====================
# 地圖類別
#=====================
class Map(pygame.sprite.Sprite):
    # 建構元(x, y, 方向, 遲延時間)
    def __init__(self, x, y):
        super().__init__()
        self.image = pygame.image.load('./images/map.png').convert_alpha()
        self.rect = self.image.get_rect()
        self.rect.topleft = (x, y)
 
    # 取得顯示子圖的左上角(x, y)及(寬度:640, 高度:640)
    def getFrame(self): 
        return 0, 0, 640, 640                                                                   
```


## 1-4. 程式 (p01.py)

``` python
#---------------------
# 匯入模組
#---------------------
import sys
import pygame
from pygame.locals import QUIT, KEYDOWN, K_UP, K_DOWN, K_LEFT, K_RIGHT, K_SPACE
from random import randint

#---------------------
# 匯入[好學生]類別
#---------------------
from ntub.m01 import Student, Map

#=====================
# 主程式
#=====================
# 啟動pygame
pygame.init()

# 設定畫面尺寸及標題
screen_surface = pygame.display.set_mode((700, 700))
pygame.display.set_caption('資一甲好學生大亂鬥')

# 產生1個好學生物件(座標x, 座標y, 方向, 更新圖片時間)    
stu = Student(10, 10, 1, 10)

# 產生地圖物件(座標x, 座標y, 方向, 更新圖片時間)    
map = Map(30, 30)

# 產生時間物件, 用來設定切換畫面頻率
clock = pygame.time.Clock()

# 遊戲主流程, 無窮迴圈
while True:
    # 偵測事件
    for event in pygame.event.get():
        # 結束
        if event.type == QUIT:
            pygame.quit()
            sys.exit()  

        # 如果按鍵
        if event.type == KEYDOWN:
            if event.key == K_UP:
                stu.turn_up() 
            elif event.key == K_DOWN:
                stu.turn_down()
            elif event.key == K_LEFT:
                stu.turn_left()  
            elif event.key == K_RIGHT:
                stu.turn_right()    
            elif event.key == K_SPACE:
                stu.turn_still()                                                          
    
    # 背景顏色, 也清除畫面
    screen_surface.fill((20, 20, 20))

    # 畫地圖
    screen_surface.blit(map.image, map.rect, map.getFrame())

    # 好學生向移動 
    stu.move()
    screen_surface.blit(stu.image, stu.rect, stu.getFrame())

    # 更新畫面
    pygame.display.update()

    # 遊戲速率(每秒畫面更新次數)
    clock.tick(100)
```
